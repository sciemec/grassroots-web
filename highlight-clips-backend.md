## FILE: app/Http/Controllers/Api/HighlightController.php

```php
<?php

namespace App\Http\Controllers\Api;

use App\Http\Controllers\Controller;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Str;

class HighlightController extends Controller
{
    /**
     * GET /players/{playerId}/highlights
     * Public — no auth required. Returns last 20 auto-generated clips for a player.
     */
    public function getPlayerHighlights(string $playerId)
    {
        $highlights = DB::table('player_highlights')
            ->where('player_id', $playerId)
            ->orderByDesc('created_at')
            ->limit(20)
            ->get();

        return response()->json(['data' => $highlights]);
    }

    /**
     * POST /highlights/callback
     * Called by Python /process-video pipeline after clips are generated and uploaded to R2.
     *
     * Expected payload:
     * {
     *   "match_id": "match_abc123",
     *   "status": "complete",
     *   "clips": [
     *     {
     *       "player_id": "uuid",
     *       "event_type": "sprint",
     *       "timestamp": 1234.5,
     *       "speed": 28.4,
     *       "url": "https://pub-xxx.r2.dev/highlights/match_abc123/clip_1.mp4",
     *       "r2_key": "highlights/match_abc123/clip_1.mp4"
     *     }
     *   ]
     * }
     */
    public function storeCallback(Request $request)
    {
        $data = $request->validate([
            'match_id'        => 'required|string|max:255',
            'status'          => 'required|string',
            'clips'           => 'required|array',
            'clips.*.player_id'  => 'required|string',
            'clips.*.event_type' => 'nullable|string',
            'clips.*.timestamp'  => 'nullable|numeric',
            'clips.*.speed'      => 'nullable|numeric',
            'clips.*.url'        => 'required|string',
            'clips.*.r2_key'     => 'required|string',
        ]);

        if ($data['status'] !== 'complete') {
            return response()->json(['message' => 'Skipped — status not complete'], 200);
        }

        $inserted = 0;
        foreach ($data['clips'] as $clip) {
            // Skip if this r2_key already exists (idempotent re-delivery)
            $exists = DB::table('player_highlights')
                ->where('r2_key', $clip['r2_key'])
                ->exists();

            if ($exists) continue;

            DB::table('player_highlights')->insert([
                'id'                => (string) Str::uuid(),
                'player_id'         => $clip['player_id'],
                'match_id'          => $data['match_id'],
                'r2_key'            => $clip['r2_key'],
                'r2_url'            => $clip['url'],
                'event_type'        => $clip['event_type'] ?? 'sprint',
                'speed_kmh'         => isset($clip['speed']) ? round((float)$clip['speed'], 2) : null,
                'timestamp_seconds' => isset($clip['timestamp']) ? round((float)$clip['timestamp'], 2) : null,
                'created_at'        => now(),
            ]);
            $inserted++;
        }

        return response()->json([
            'message'  => "Stored {$inserted} highlight clip(s) for match {$data['match_id']}",
            'inserted' => $inserted,
        ], 201);
    }
}
```
